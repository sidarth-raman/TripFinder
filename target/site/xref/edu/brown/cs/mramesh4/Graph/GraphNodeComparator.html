3   import java.util.Comparator;
4   
5   /**
6    * This class is a custom comparator for our priority queue used in the Graph class.
7    * It takes in any type of Node that implements GraphNode.
8    * @param <N> representing type of Node we want to compare
9    * @param <E> representing type of Edge we want to compare
10   */
11  public class GraphNodeComparator<N extends GraphNode<N, E>, E extends GraphEdge<N, E>>
12          implements Comparator<N> {
13  
14    @Override
15    public int compare(N a, N b) {
16      //compare total weight (distance to node + distance to goal)
17      return Double.compare(a.getTotalWeight(), b.getTotalWeight());
18    }
19  
20  }
