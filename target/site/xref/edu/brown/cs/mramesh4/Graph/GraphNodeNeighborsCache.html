3   import com.google.common.cache.CacheBuilder;
4   import com.google.common.cache.CacheLoader;
5   import com.google.common.cache.LoadingCache;
6   import java.util.List;
7   import java.util.concurrent.ExecutionException;
8   import java.util.concurrent.TimeUnit;
9   
10  /**
11   * This class represents a cache used in the Graph class for A* to improve
12   * runtime. It maps Nodes to a list of their neighbors.
13   * @param <N> representing type of Node being used in graph.
14   * @param <E> representing type of Edge being used in graph.
15   */
16  public class GraphNodeNeighborsCache<N extends GraphNode<N, E>, E extends GraphEdge<N, E>> {
17  
18    private LoadingCache<N, List<N>> cache;
19  
20    /**
21     * In the constructor for this class, we initialize our cache.
22     */
23    public GraphNodeNeighborsCache() {
24      cache = CacheBuilder.newBuilder()
25        .refreshAfterWrite(2, TimeUnit.SECONDS)
26        .build(new CacheLoader<N, List<N>>() {
27          @Override
28          public List<N> load(N key) throws Exception {
29            return key.neighbors();
30          }
31        });
32    }
33  
34    /**
35     * This method gets the neighbors of the node passed in from the cache.
36     * @param key representing current Node we are
37     * @return the neighbors of the Node.
38     * @throws ExecutionException if our cache cannot execute the query
39     */
40    public List<N> getNode(N key) throws ExecutionException {
41      return cache.get(key);
42    }
43  
44  }
