451   /**.
452    * Calculates the k nearest neighbors to a point in space
453    * @param k a non-negative int representing number of neighbors
454    * @param x  a double representing our point's x value
455    * @param y a double representing our point's y value
456    * @param z  a double representing our point's z value
457    * @return a List of Stars closest to the point
458    */
459   public List<Star> naiveNearestNeighbors(int k, double x, double y, double z) {
460     ArrayList<Star> ret = new ArrayList<>();
461     if (k == 0) {
462       return ret;
463     }
464     //get the star we want to use
465     Star toFind = findStar(x, y, z);
466 
467     //if the star doesn't exist, no neighbors exist
468     if (toFind == null) {
469       System.err.println("ERROR: There is no star with these coordinates ");
470       return null;
471     }
472 
473     //create a hashmap that stores a list of doubles and stars with the distance
474     HashMap<Double, ArrayList<Star>> map = new HashMap<>();
475 
476     //add each star into a hashmap based on its distance ;
477     for (Star star: starList) {
478       double distance = star.calculateDistanceFromStar(toFind);
479       if (!map.containsKey(distance)) {
480         map.put(distance, new ArrayList<Star>());
481       }
482       map.get(distance).add(star);
483     }
484 
485     //sort the entries into a list of keys;
486     ArrayList<Double> keys = new ArrayList<>(map.keySet());
487     //sort the keys in order from descending to ascending
488     Collections.sort(keys);
489     //a counter to go through the list
490     int start = 0;
491     //while we still have keys to go through and more neighbors:
492     while (start < keys.size() && k > 0) {
493       //get the nextClosestStars
494       ArrayList<Star> nextClosestStars = map.get(keys.get(start));
495       //subtract k
496       int size = nextClosestStars.size();
497       k -= size;
498       //if k is below 0, that means there was
499       // overflow by the size - k, so we need to select
500       //size - k elements to add
501       if (k < 0) {
502         for (int i = 0; i < size - Math.abs(k); i++) {
503           //generate a randomInt based on the current remaining stars
504           int randomInt = (int) (Math.random()
505               * (nextClosestStars.size()));
506           //add the star
507           ret.add(nextClosestStars.get(randomInt));
508           //remove it from the list of remaining stars
509           nextClosestStars.remove(randomInt);
510         }
511         //since we have hit k we are done;
512         break;
513       } else {
514         //add the nextStar
515         for (Star nextStar: nextClosestStars) {
516           ret.add(nextStar);
517         }
518         //increment start;
519         start++;
520       }
521     }
522     //return our arrayList
523     return ret;
524   }
