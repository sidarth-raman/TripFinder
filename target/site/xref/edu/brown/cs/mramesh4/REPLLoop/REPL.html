3   import edu.brown.cs.mramesh4.stars.ActionMethod;
4   import java.io.InputStreamReader;
5   import java.util.HashMap;
6   
7   /**
8    * This class represents a REPL.
9    */
10  public class REPL {
11  
12    private Parser inputParser;
13    private HashMap<String, ActionMethod<?>> commands;
14  
15    /**
16     * This is a method to create a REPL.
17     * @param commands map of commands to run
18     */
19    public REPL(HashMap<String, ActionMethod<?>> commands) {
20      this.commands = commands;
21    }
22  
23    /**
24     * This is a method to read the lines of the REPL.
25     */
26    public void read() {
27  
28      inputParser = new Parser(new InputStreamReader(System.in));
29  
30      while (true) {
31  
32        // parses user input
33        String[] input = inputParser.parse();
34  
35        // EOF condition
36        if (input == null) {
37          break;
38        }
39  
40        if (commands.containsKey(input[0])) {
41          commands.get(input[0]).run(input);
42          //command doesn't exist, throw ERROR
43        } else {
44          System.err.println("ERROR: Command does not exist.");
45        }
46  
47      }
48    }
49  }
