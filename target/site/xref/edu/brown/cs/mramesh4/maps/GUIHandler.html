44    /**
45     * This is a GuiHandler method used to run route methods.
46     * @param call name of method to run
47     * @return a handler
48     */
49    public Route method(String call) {
50      Route c = null;
51      switch (call) {
52        case "UserHandler":
53          c =  new UserHandler();
54          break;
55        case "WayHandler":
56          c = new WayHandler();
57          break;
58        case "RouteCoorsHandler":
59          c = new RouteCoorsHandler();
60          break;
61        case "NearestHandler":
62          c = new NearestHandler();
63          break;
64        case "RouteStreetsHandler":
65          c = new RouteStreetsHandler();
66          break;
67        case "CheckInHandler":
68          c = new CheckInHandler();
69          break;
70        default:
71          c = null;
72          break;
73      }
74      return c;
75    }

183         return GSON.toJson(variables);

231   /**
232    * Handles the radius call on the gui.
233    */
234   private static class RadiusHandler implements TemplateViewRoute {
235     @Override
236     public ModelAndView handle(Request req, Response res) {
237       QueryParamsMap qm = req.queryMap();
238       String cmd = qm.value("text2");
239       String ret = "No results to display";
240       if (db.getStarList().size() == 0) {
241         ret = "Please load a file in the command line";
242       } else if (cmd == null || cmd.equals("")) {
243         ret = "Please input a valid answer";
244       } else {
245         String neighbs = qm.value("radius_switched");
246         String start = "naive_radius";
247         if (neighbs != null) {
248           start = "radius";
249         }
250         String cmd2 = start.concat(" ").concat(cmd);
251         String[] coms = cmd2.split(" ");
252         List<Star> stars = db.radiusRet(cmd2, coms);
253 
254         if (db.getStarList().size() == 0) {
255           ret = "Please load a file in the command line";
256         }
257         if (coms[0].toUpperCase().equals("NAIVE_NEIGHBORS")
258             || coms[0].toUpperCase().equals("NEIGHBORS")) {
259           ret = "Please use the neighbors box for this action";
260         } else {
261           if (db.getStarList().size() == 0) {
262             ret = "Please load a file in the command line";
263           }
264           if (stars != null && !stars.isEmpty()) {
265             ret = "The stars within the radius: " + coms[1] + " are";
266             ret += "<br> <br>";
267             ret += "<table id=\"table\">"
268               + "<tr> <th> Match # </th> <th> Name </th>"
269               + "<th> ID </th> <th> X </th> <th> Y </th> <th> Z </th> </tr>";
270             for (int i = 0; i < stars.size(); i++) {
271               Star curr = stars.get(i);
272               ret += "<tr><td>" + Integer.toString(i + 1) + "</td>" + "<td>" + curr.getName()
273                 + "</td>" + "<td>" + curr.getStarID() + "</td>"
274                 + "<td>" + curr.getX() + "</td>"
275                 + "<td>" + curr.getY() + "</td>"
276                 + "<td>" + curr.getZ() + "</td>";
277               ret += "</tr>";
278             }
279             ret += "</table>";
280           } else {
281             ret = "No stars found in the search";
282           }
283         }
284       }
285       Map<String, Object> variables = ImmutableMap.of("title",
286           "Stars: Query the database", "results", ret);
287       return new ModelAndView(variables, "query.ftl");
288     }
