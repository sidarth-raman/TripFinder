71    /**
72     * Gets the distance between two WayNode objects.
73     * @param a the waynode object to compare distance with
74     * @return double representing euclidean distance between the two objects
75     */
76    public double distanceBetween(Dimensional a) {
77      if (a instanceof WayNodes) {
78        double aX = a.getCoordinate(0);
79        double aY = a.getCoordinate(1);
80        return Math.sqrt(Math.pow(Math.abs((aX - this.longitude)), 2)
81          + Math.pow(Math.abs((aY - this.lat)), 2));
82      } else {
83        return -1;
84      }
85    }

238   /**
239    * This method calculates the distance to the goal using Haversine distance.
240    * @param goal representing goal node
241    * @return double representing distance.
242    */
243   @Override
244   public double toGoal(WayNodes goal) {
245     double thisLat = Math.toRadians(lat);
246     double thisLon = Math.toRadians(longitude);
247     double goalLat = Math.toRadians(goal.getLat());
248     double goalLon = Math.toRadians(goal.getLong());
249     /**
250      * Haversine formula: I got this from
251      * https://www.geeksforgeeks.org/program-distance-two-points-earth/
252      */
253     double dlon = goalLon - thisLon;
254     double dlat = goalLat - thisLat;
255     double havernmath = Math.pow(Math.sin(dlat / 2), 2)
256             + Math.cos(thisLat) * Math.cos(goalLat)
257             * Math.pow(Math.sin(dlon / 2), 2);
258     double finalanswer = 2 * Math.asin(Math.sqrt(havernmath));
259     // Radius of earth in kilometers. Use 3956 for miles
260     return (finalanswer * EARTH_RADIUS_IN_KM);
261   }
