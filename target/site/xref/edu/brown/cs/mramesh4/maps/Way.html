43    /**
44     * Calculates the distance between two waynodes objects.
45     * The code for this uses the haversine formula from:
46     * https://www.geeksforgeeks.org/program-distance-two-points-earth/
47     * @param a first Waynode
48     * @param b second Waynode
49     * @return double representing haversine distance
50     */
51    public double getHaversineDistance(WayNodesref="../../../../../edu/brown/cs/mramesh4/maps/WayNodes.html#WayNodes">WayNodes a, WayNodes b) {
52      if (a == null || b == null) {
53        return -1;
54      }
55      double lat1 = Math.toRadians(a.getLat());
56      double lon1 = Math.toRadians(a.getLong());
57      double lat2 = Math.toRadians(b.getLat());
58      double lon2 = Math.toRadians(b.getLong());
59      /**
60       * Haversine formula: I got this from
61       * https://www.geeksforgeeks.org/program-distance-two-points-earth/
62       */
63      double dlon = lon2 - lon1;
64      double dlat = lat2 - lat1;
65      double havernmath = Math.pow(Math.sin(dlat / 2), 2)
66              + Math.cos(lat1) * Math.cos(lat2)
67              * Math.pow(Math.sin(dlon / 2), 2);
68      double finalanswer = 2 * Math.asin(Math.sqrt(havernmath));
69      // Represents radius of earth in kilometers. Use 3956 for miles
70      return (finalanswer * EARTH_RADIUS_IN_KM);
71    }
