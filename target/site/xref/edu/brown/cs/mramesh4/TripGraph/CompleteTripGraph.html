124   /**
125    * Helper method to conduct the AStar search recursively and use backtracking
126    * to save the route, since this method could not be used previously.
127    * @param end the end node to search for
128    * @param pq priorityqueue to use
129    * @param nodes nodes of curr path
130    * @param visited map of visited
131    * @return a list of Nodes.
132    */
133   public List<N> aStarHelp(N end, PriorityQueue<N> pq, List<N> nodes, HashMap<String, N> visited){
134     if(!pq.isEmpty()){
135       N curr = pq.poll();
136       nodes.add(curr);
137       if (curr.equals(end)) {
138         //return a copy in order to prevent alteration
139         return new ArrayList<>(nodes);
140       }
141       List<N> neighbors = curr.getNeighbors();
142       for(N next: neighbors){
143         if(next.equals(curr)) {
144           continue;
145         } else{
146           String name = next.getName();
147           Double dist = curr.getDistance() + curr.getConnectingEdges().get(name).getWeight();
148           Double totalWeight = dist + next.toGoal(end);
149           if (!visited.containsKey(next.getName())) {
150             next.setDistance(dist);
151             next.setWeight(totalWeight);
152             visited.put(next.getName(), next);
153             pq.add(next);
154           } else {
155             N prev = visited.get(next.getName());
156             //checking if we want to update the weight and distance
157             if (prev.getWeight() > totalWeight) {
158               next.setDistance(dist);
159               next.setWeight(totalWeight);
160               visited.put(next.getName(), next);
161               pq.add(next);
162             }
163           }
164         }
165       }
166       //backtracking
167       List<N> ret = aStarHelp(end, pq, nodes, visited);
168       nodes.remove(curr);
169       return ret;
170     } else{
171       if(visited.containsKey(end)){
172         return nodes;
173       } else{
174         return null;
175       }
176     }
177   }
