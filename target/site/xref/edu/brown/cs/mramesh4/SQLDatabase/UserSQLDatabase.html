22    /**
23     * This is a constructor to access our database.
24     * @param datafile This is the datafile to access
25     */
26    public UserSQLDatabase(String datafile)  {
27      try {
28        url = datafile;
29        setConnection();
30        Statement stat = conn.createStatement();
31        stat.executeUpdate("PRAGMA foreign_keys=ON;");
32        stat.executeUpdate("CREATE TABLE IF NOT EXISTS users ("
33            + "id INTEGER NOT NULL,"
34            + "name VARCHAR(255) NOT NULL, "
35            + "ts FLOAT NOT NULL,"
36            + "lat FLOAT NOT NULL,"
37            + "lon FLOAT NOT NULL);");
38        stat.close();
39      } catch (SQLException e) {
40        System.err.println(e);
41      }
42  
43    }

54     * This is a method to set a connection our database.
55     */
56    public void setConnection() {
57      try {
58        Class.forName("org.sqlite.JDBC");
59        String urlToDB = "jdbc:sqlite:" + url;
60        conn = DriverManager.getConnection(urlToDB);
61        //return conn;
62      } catch (ClassNotFoundException | SQLException e) {
63        System.err.println(e);
64      }
65    }

67    /**
68     * This is a method to add a user to the database.
69     * @param userData this is a List of parameters to put on the
70     */
71    public void addUser(List<String> userData) {
72      try {
73        PreparedStatement insertionPrep = conn.prepareStatement("INSERT OR IGNORE INTO users "
74            + "(id, name, ts, lat, lon)"
75            + "VALUES (?, ?, ?, ?, ?)");
76        int id = Integer.parseInt(userData.get(0));
77        String name = userData.get(1);
78        Double big = new BigDecimal(userData.get(2)).doubleValue();
79        Double lat = Double.parseDouble(userData.get(3));
80        Double lon = Double.parseDouble(userData.get(4));
81        insertionPrep.setInt(1, id);
82        insertionPrep.setString(2, name);
83        insertionPrep.setDouble(3, big);
84        insertionPrep.setDouble(4, lat);
85        insertionPrep.setDouble(5, lon);
86        insertionPrep.executeUpdate();
87        insertionPrep.close();
88      } catch (NumberFormatException | ArrayIndexOutOfBoundsException | SQLException e) {
89        System.err.println(e);
90      }
91    }

93    /**.
94     * This is our method to delete a user.
95     * @param id This is the id of the user to delete
96     * @throws SQLException exception thrown if problems deleting
97     */
98    public void deleteUser(String id) throws SQLException {
99      PreparedStatement deletionPrep = conn.prepareStatement("DELETE FROM users WHERE id = ?");
100     deletionPrep.setString(1, id);
101     deletionPrep.executeUpdate();
102     deletionPrep.close();
103   }

105   /**
106    * This method gets all a users checkins.
107    * @param id the id of the user
108    * @return a list of checkins
109    * @throws SQLException sql exception
110    */
111   public List<UserCheckin> getUserInfo(String id) throws SQLException {
112     PreparedStatement queryPrep = conn.prepareStatement("SELECT * FROM users WHERE id = ?");
113     queryPrep.setString(1, id);
114     ResultSet rs = queryPrep.executeQuery();
115     List<UserCheckin> ret = new ArrayList<>();
116     while (rs.next()) {
117       UserCheckin curr;
118       try {
119         int getID = rs.getInt(1);
120         String getName = rs.getString(2);
121         Double ts = rs.getBigDecimal(3).doubleValue();
122         Double lat = rs.getDouble(4);
123         Double lon = rs.getDouble(5);
124         curr = new UserCheckin(getID, getName, ts, lat, lon);
125         ret.add(curr);
126       } catch (NumberFormatException | SQLException | NullPointerException e) {
127         System.err.println("Can't get user info");
128       }
129     }
130     rs.close();
131     queryPrep.close();
132     return ret;
133   }
