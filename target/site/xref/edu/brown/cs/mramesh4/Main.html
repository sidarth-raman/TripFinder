176   private static class ExceptionPrinter implements ExceptionHandler {
177     @Override
178     public void handle(Exception e, Request req, Response res) {
179       res.status(500);
180       StringWriter stacktrace = new StringWriter();
181       try (PrintWriter pw = new PrintWriter(stacktrace)) {
182         pw.println("<pre>");
183         e.printStackTrace(pw);
184         pw.println("</pre>");
185       }
186       res.body(stacktrace.toString());
187     }
